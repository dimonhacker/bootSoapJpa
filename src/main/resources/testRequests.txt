<!--Поиск по логину-->
<soapenv:Envelope xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/"
                  xmlns:gs="http://localhost">
    <soapenv:Header/>
    <soapenv:Body>
        <gs:getUserRequest>
            <gs:login>login</gs:login>
        </gs:getUserRequest>
    </soapenv:Body>
</soapenv:Envelope>

<soapenv:Envelope xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/"

<soapenv:Body>
    <gs:delUserRequest>
        <gs:login>login</gs:login>
    </gs:delUserRequest>
</soapenv:Body>
</soapenv:Envelope>
